67 template <typename T>
68 void writeFloat(T value) {
69 if (isnan(value))
70 return writeRaw(ARDUINOJSON_ENABLE_NAN ? "NaN" : "null");
71
72#if ARDUINOJSON_ENABLE_INFINITY
73 if (value < 0.0) {
74 writeRaw('-');
75 value = -value;
76 }
77
78 if (isinf(value))
79 return writeRaw("Infinity");
80#else
81 if (isinf(value))
82 return writeRaw("null");
83
84 if (value < 0.0) {
85 writeRaw('-');
86 value = -value;
87 }
88#endif
89
90 FloatParts<T> parts(value);
91
92 writeInteger(parts.integral);
93 if (parts.decimalPlaces)
94 writeDecimals(parts.decimal, parts.decimalPlaces);
95
96 if (parts.exponent) {
97 writeRaw('e');
98 writeInteger(parts.exponent);
99 }
100 }

102 template <typename T>
103 typename enable_if<is_signed<T>::value>::type writeInteger(T value) {
104 typedef typename make_unsigned<T>::type unsigned_type;
105 unsigned_type unsigned_value;
106 if (value < 0) {
107 writeRaw('-');
108 unsigned_value = unsigned_type(unsigned_type(~value) + 1);
109 } else {
110 unsigned_value = unsigned_type(value);
111 }
112 writeInteger(unsigned_value);
113 }

115 template <typename T>
116 typename enable_if<is_unsigned<T>::value>::type writeInteger(T value) {
117 char buffer[22];
118 char* end = buffer + sizeof(buffer);
119 char* begin = end;
120
121 // write the string in reverse order
122 do {
123 *--begin = char(value % 10 + '0');
124 value = T(value / 10);
125 } while (value);
126
127 // and dump it in the right order
128 writeRaw(begin, end);
129 }

131 void writeDecimals(uint32_t value, int8_t width) {
132 // buffer should be big enough for all digits and the dot
133 char buffer[16];
134 char* end = buffer + sizeof(buffer);
135 char* begin = end;
136
137 // write the string in reverse order
138 while (width--) {
139 *--begin = char(value % 10 + '0');
140 value /= 10;
141 }
142 *--begin = '.';
143
144 // and dump it in the right order
145 writeRaw(begin, end);
146 }
