29// begin_ end_
30// v v
31// +-------------+--------------+--------------+
32// | strings... | (free) | ...variants |
33// +-------------+--------------+--------------+
34// ^ ^
35// left_ right_

37class MemoryPool {
38 public:
39 MemoryPool(char* buf, size_t capa)
40 : begin_(buf),
41 left_(buf),
42 right_(buf ? buf + capa : 0),
43 end_(buf ? buf + capa : 0),
44 overflowed_(false) {
45 ARDUINOJSON_ASSERT(isAligned(begin_));
46 ARDUINOJSON_ASSERT(isAligned(right_));
47 ARDUINOJSON_ASSERT(isAligned(end_));
48 }

72 template <typename TAdaptedString>
73 const char* saveString(TAdaptedString str) {
74 if (str.isNull())
75 return 0;
76
77#if ARDUINOJSON_ENABLE_STRING_DEDUPLICATION
78 const char* existingCopy = findString(str);
79 if (existingCopy)
80 return existingCopy;
81#endif
82
83 size_t n = str.size();
84
85 char* newCopy = allocString(n + 1);
86 if (newCopy) {
87 stringGetChars(str, newCopy, n);
88 newCopy[n] = 0; // force null-terminator
89 }
90 return newCopy;
91 }

135 // Squash the free space between strings and variants
136 //
137 // begin_ end_
138 // v v
139 // +-------------+--------------+
140 // | strings... | ...variants |
141 // +-------------+--------------+
142 // ^
143 // left_ right_
144 //
145 // This funcion is called before a realloc.
146 ptrdiff_t squash() {
147 char* new_right = addPadding(left_);
148 if (new_right >= right_)
149 return 0;
150
151 size_t right_size = static_cast<size_t>(end_ - right_);
152 memmove(new_right, right_, right_size);
153
154 ptrdiff_t bytes_reclaimed = right_ - new_right;
155 right_ = new_right;
156 end_ = new_right + right_size;
157 return bytes_reclaimed;
158 }
159
160 // Move all pointers together
161 // This funcion is called after a realloc.
162 void movePointers(ptrdiff_t offset) {
163 begin_ += offset;
164 left_ += offset;
165 right_ += offset;
166 end_ += offset;
167 }

169 private:
170 void checkInvariants() {
171 ARDUINOJSON_ASSERT(begin_ <= left_);
172 ARDUINOJSON_ASSERT(left_ <= right_);
173 ARDUINOJSON_ASSERT(right_ <= end_);
174 ARDUINOJSON_ASSERT(isAligned(right_));
175 }

177#if ARDUINOJSON_ENABLE_STRING_DEDUPLICATION
178 template <typename TAdaptedString>
179 const char* findString(const TAdaptedString& str) const {
180 size_t n = str.size();
181 for (char* next = begin_; next + n < left_; ++next) {
182 if (next[n] == '\0' && stringEquals(str, adaptString(next, n)))
183 return next;
184
185 // jump to next terminator
186 while (*next)
187 ++next;
188 }
189 return 0;
190 }
191#endif
192
193 char* allocString(size_t n) {
194 if (!canAlloc(n)) {
195 overflowed_ = true;
196 return 0;
197 }
198 char* s = left_;
199 left_ += n;
200 checkInvariants();
201 return s;
202 }

204 template <typename T>
205 T* allocRight() {
206 return reinterpret_cast<T*>(allocRight(sizeof(T)));
207 }
208
209 void* allocRight(size_t bytes) {
210 if (!canAlloc(bytes)) {
211 overflowed_ = true;
212 return 0;
213 }
214 right_ -= bytes;
215 return right_;
216 }

218 char *begin_, *left_, *right_, *end_;
219 bool overflowed_;

222template <typename TAdaptedString, typename TCallback>
223bool storeString(MemoryPool* pool, TAdaptedString str,
224 StringStoragePolicy::Copy, TCallback callback) {
225 const char* copy = pool->saveString(str);
226 JsonString storedString(copy, str.size(), JsonString::Copied);
227 callback(storedString);
