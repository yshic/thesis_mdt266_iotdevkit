13class NestingLimit {

16 explicit NestingLimit(uint8_t n) : value_(n) {}
17
18 NestingLimit decrement() const {
19 ARDUINOJSON_ASSERT(value_ > 0);
20 return NestingLimit(static_cast<uint8_t>(value_ - 1));
