17#define PID_MAX_INTEGRAL 2000
18#define ZUMO_BASE_DEADBAND 20
19
20// RC-servo values
21#define PIXY_RCS_MIN_POS 0
22#define PIXY_RCS_MAX_POS 1000L
23#define PIXY_RCS_CENTER_POS ((PIXY_RCS_MAX_POS-PIXY_RCS_MIN_POS)/2)
24
25class PIDLoop
26{
27public:
28 PIDLoop(int32_t pgain, int32_t igain, int32_t dgain, bool servo)
29 {
30 m_pgain = pgain;
31 m_igain = igain;
32 m_dgain = dgain;
33 m_servo = servo;
34
35 reset();
36 }
37
38 void reset()
39 {
40 if (m_servo)
41 m_command = PIXY_RCS_CENTER_POS;
42 else
43 m_command = 0;
44
45 m_integral = 0;
46 m_prevError = 0x80000000L;
47 }
48
49 void update(int32_t error)
50 {
51 int32_t pid;
52
53 if (m_prevError!=0x80000000L)
54 {
55 // integrate integral
56 m_integral += error;
57 // bound the integral
58 if (m_integral>PID_MAX_INTEGRAL)
59 m_integral = PID_MAX_INTEGRAL;
60 else if (m_integral<-PID_MAX_INTEGRAL)
61 m_integral = -PID_MAX_INTEGRAL;
62
63 // calculate PID term
64 pid = (error*m_pgain + ((m_integral*m_igain)>>4) + (error - m_prevError)*m_dgain)>>10;
65
66 if (m_servo)
67 {
68 m_command += pid; // since servo is a position device, we integrate the pid term
69 if (m_command>PIXY_RCS_MAX_POS)
70 m_command = PIXY_RCS_MAX_POS;
71 else if (m_command<PIXY_RCS_MIN_POS)
72 m_command = PIXY_RCS_MIN_POS;
73 }
74 else
75 {
76 // Deal with Zumo base deadband
77 if (pid>0)
78 pid += ZUMO_BASE_DEADBAND;
79 else if (pid<0)
80 pid -= ZUMO_BASE_DEADBAND;
81 m_command = pid; // Zumo base is velocity device, use the pid term directly
82 }
83 }
84
85 // retain the previous error val so we can calc the derivative
86 m_prevError = error;
87 }
88
89 int32_t m_command;
90
91private:
92 int32_t m_pgain;
93 int32_t m_igain;
94 int32_t m_dgain;
95
96 int32_t m_prevError;
97 int32_t m_integral;
98 bool m_servo;
99};
