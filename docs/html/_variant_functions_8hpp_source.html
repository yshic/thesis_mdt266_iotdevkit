96inline size_t variantNesting(const VariantData* var) {
97 if (!var)
98 return 0;
99
100 const CollectionData* collection = var->asCollection();
101 if (!collection)
102 return 0;
103
104 size_t maxChildNesting = 0;
105 for (const VariantSlot* s = collection->head(); s; s = s->next()) {
106 size_t childNesting = variantNesting(s->data());
107 if (childNesting > maxChildNesting)
108 maxChildNesting = childNesting;
109 }
110 return maxChildNesting + 1;
111}
