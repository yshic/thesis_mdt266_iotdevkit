7/*
8 This struct helps us constructing the I2C output based on data and control outputs.
9 Because the LCD is set to 4-bit mode, 4 bits of the I2C output are for the control outputs
10 while the other 4 bits are for the 8 bits of data which are send in parts using the enable output.
11*/

14struct OutputState
15{
16 // Select register
17 // 0: instruction register
18 // 1: data register
19 uint8_t rs = 0;
20 // select read or write
21 uint8_t rw = 0;
22 // Enable: starts data read or write
23 uint8_t en = 0;
24 // LED status
25 uint8_t led = 0;
26 uint8_t data = 0;
27
28 uint8_t GetLowData()
29 {
30 uint8_t buffer = rs;
31 buffer |= rw << 1;
32 buffer |= en << 2;
33 buffer |= led << 3;
34 buffer |= (data & 0x0F) << 4;
35 return buffer;
36 }
37
38 uint8_t GetHighData()
39 {
40 uint8_t buffer = rs;
41 buffer |= rw << 1;
42 buffer |= en << 2;
43 buffer |= led << 3;
44 buffer |= (data & 0xF0);
45 return buffer;
46 }
47};
